```python
from sklearn.svm import *
import numpy as np 
svr = SVR(C=2.0, epsilon=0.2)
data_file = open('./dataset.txt')
label_file = open('./dataset1.txt')

data = data_file.read().split('\n')
data = np.array([i.split() for i in data])
data = data.astype(np.float64)
label = label_file.read().split('\n')
label = np.array(label)
label = label.astype(np.float64)

svr.fit(data, label)
print(svr.support_vectors_)
y_pred = svr.predict(data)
print(y_pred)
```
